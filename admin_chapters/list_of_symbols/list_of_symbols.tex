% The longtable environment allows the table to span multiple pages
\begin{longtable}[l]{l l}
    $A$             & Arguments to the function-under-test                                       \\
    $E$             & Exceptions thrown by the function during execution                         \\
    $f$             & A function, treated as a transformation $f: (A, S) \rightarrow (R, E, S')$ \\
    $m$             & A method under consideration (Elbaum et al.\ formalism)                    \\
    $p$             & $p$-value of a statistical test                                            \\
    $R$             & Return value of the function-under-test                                    \\
    $R^2$           & Coefficient of determination                                               \\
    $S$             & External state before the function executes                                \\
    $S'$            & External state after the function executes                                 \\
    $T_\text{exec}$ & Unit test execution time, excluding test collection overhead               \\
    $T_\text{real}$ & Total unit test runtime, including test collection overhead                \\
    $T_\text{sys}$  & System test execution time                                                 \\
    $U$             & Mann-Whitney $U$ statistic                                                 \\
\end{longtable}
